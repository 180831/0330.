//懒汉式(单例模式)
public class Test2 {

    private static Test2 test2;

    private Test2(){
        System.out.println("This is a pig");
    }
    public static Test2 fun(){
        if(test2 == null){
            test2 = new Test2();
        }
        return test2;
    }

    public void fun2(){
        System.out.println("This is a dog");
    }

    public static void main(String[] args) {
        Test2 test2 = Test2.fun();

        test2.fun2();
    }
}
